import mathdef round_order_amounts(price, size):    # Round price to tick size (e.g., 0.01)    price = round(price, 2)        # For BUY: makerAmount = price * size (USDC, 2 decimals)    maker_amount = math.floor(price * size * 100) / 100  # 2 decimals        # takerAmount = size (shares, 4 decimals)      taker_amount = math.floor(size * 10000) / 10000  # 4 decimals        return price, taker_amount# Exampleprice, size = round_order_amounts(0.53, 5.66)
